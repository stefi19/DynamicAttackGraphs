\begin{frame}{Executive summary}
  \begin{itemize}
    \item Problem: full recomputation of attack graphs is costly for live networks
    \item Approach: use differential dataflow (Timely) to incrementally update derived facts
    \item Artifacts: Rust implementation, benchmarks (chain/star/random cut), Graphviz visualisations
    \item Results: large speedups on star/topologies; position-dependent cost on chain
  \end{itemize}
  \note{Give a brief overview: problem, technique, implementation, and headline results.}

\begin{frame}{Motivation}
  \begin{itemize}
    \item Attack graphs summarise attacker reachability using facts and rules
    \item Modern networks are dynamic: CVEs, patches, topology changes
    \item Recomputing the entire graph on every change is impractical at scale
  \end{itemize}
  \note{Give examples where rapid updates are needed: patch management, red-team, live monitoring.}
\end{frame}

\begin{frame}{Background: attack graphs and MulVAL}
  \begin{itemize}
    \item Attack graphs: facts (vulnerabilities, network edges, firewall rules) and inference rules
    \item MulVAL uses Datalog-like rules to compute derived reachability facts
    \item Example rule (informal): execCode(A,D,P) :- execCode(A,Src,_), effectiveAccess(Src,D,Svc), vulnerability(D,_,Svc,P)
  \end{itemize}
  \note{Explain the Datalog style rules and how they map to joins and recursive definitions.}
\end{frame}

\begin{frame}{Background: differential dataflow}
  \begin{itemize}
    \item Differential dataflow propagates deltas and supports iterative fixed-point computation
    \item Operators: join, antijoin, iterate, distinct, consolidate
    \item Natural fit for incremental Datalog: update deltas, recompute only affected facts
  \end{itemize}
  \note{Highlight why iterate() + distinct() yields termination and how antijoin handles negation (firewall denies).}
\end{frame}

  \end{itemize}

\begin{frame}{Derived facts and rules}
  \begin{itemize}
    \item \texttt{EffectiveNetworkAccess(src,dst,svc)} = network \textbf{ANTIJOIN} denies
    \item \texttt{AttackerCodeExecution(attacker,host,priv)} formed by recursion via \texttt{iterate()}
    \item \texttt{AttackerOwnsMachine(attacker,host)} when \texttt{priv==Root}
    \item \texttt{AttackerGoalReached(attacker,target)} via semijoin on owned machines
  \end{itemize}
  \note{Explicitly connect each derived fact to the operator chain used in the code.}
\end{frame}

\begin{frame}{Translating rules to dataflow (example)}
  \begin{itemize}
    \item effectiveAccess = network_rules.antijoin(deny_keys)
    \item execCode: iterate over current execCode; join with access (by src); join with vulnerabilities (by host,service); concat + distinct
  \end{itemize}
  \note{Show how join keys are chosen and why distinct() is necessary to ensure termination.}
\end{frame}
